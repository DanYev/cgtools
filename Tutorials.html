This document demonstrates the Gmx Pipe pipeline used in reForge for setting up and running coarse-grained MD simulations with GROMACS. Below is the full source code for the pipeline module.

  1#!/usr/bin/env python
  2"""
  3Gmx Pipe Tutorial Module
  4
  5This module serves as a tutorial for setting up and running coarse-grained
  6molecular dynamics (MD) simulations using the reForge package and GROMACS. It
  7provides a pipeline that performs several key tasks:
  8  1. System Setup: Copying force field and parameter files, preparing directories,
  9     cleaning and sorting the input PDB file, and splitting chains.
 10  2. Coarse-Graining: Applying coarse-graining methods for proteins (using both
 11     Go-model and elastic network approaches) and nucleotides.
 12  3. Solvation and Ion Addition: Solvating the system and adding ions to neutralize it.
 13  4. MD Simulation: Running energy minimization, equilibration, and production MD runs.
 14  5. Post-Processing: Generating index files, converting trajectories, and performing
 15     analyses such as RMSF, RMSD, covariance analysis, clustering, and time-dependent
 16     correlation calculations.
 17
 18Usage Example:
 19    To set up a coarse-grained protein/RNA system:
 20      >>> from gmx_pipe import setup
 21      >>> sysdir = "/path/to/simulation"
 22      >>> sysname = "my_system"
 23      >>> setup(sysdir, sysname)
 24
 25    To run an MD simulation:
 26      >>> from gmx_pipe import md
 27      >>> runname = "run1"
 28      >>> ntomp = 8
 29      >>> md(sysdir, sysname, runname, ntomp)
 30
 31Author: DY
 32Date: 2025-XX-XX
 33"""

 60def setup_cg_protein_rna(sysdir, sysname):
 61    """Sets up a coarse-grained protein/RNA system.
 62
 63    This function performs the following steps:
 64      1. Prepares system files and directories.
 65      2. Sorts and cleans the input PDB file.
 66      3. Splits chains and cleans them using GROMACS.
 67      4. Retrieves GO contact maps.
 68      5. Applies coarse-graining (using the Go-model for proteins and Martini RNA for nucleotides).
 69      6. Creates topology and structure files.
 70      7. Solvates the system and adds ions.
 71      8. Generates index files.
 72
 73    Args:
 74        sysdir (str): Base directory for the system.
 75        sysname (str): Name of the MD system.
 76    """
 77    mdsys = gmxSystem(sysdir, sysname)
 78    # 1. Prepare force field, parameter files, and directories.
 79    mdsys.prepare_files()
 80    mdsys.sort_input_pdb(f"{sysname}.pdb")
 81
 82    # 2. Clean the PDB and split chains using GROMACS.
 83    mdsys.clean_pdb_gmx(in_pdb=mdsys.inpdb, clinput="8\n 7\n", ignh="no", renum="yes")
 84    mdsys.split_chains()
 85    mdsys.clean_chains_gmx(clinput="8\n 7\n", ignh="yes", renum="yes")
 86    mdsys.get_go_maps(append=True)
 87
 88    # 3. Coarse-grain the proteins and RNA.
 89    mdsys.martinize_proteins_go(go_eps=10.0, go_low=0.3, go_up=1.0, p="backbone", pf=500, append=False)
 90    mdsys.martinize_rna(ef=200, el=0.3, eu=1.2, p="backbone", pf=500, append=True)
 91    mdsys.make_martini_topology_file(add_resolved_ions=False, prefix="chain")
 92    mdsys.make_cgpdb_file(bt="dodecahedron", d="1.2")
 93    
 94    # 4. Solvate and add ions.
 95    solvent = os.path.join(mdsys.wdir, "water.gro")
 96    mdsys.solvate(cp=mdsys.solupdb, cs=solvent)
 97    mdsys.add_bulk_ions(conc=0.15, pname="NA", nname="CL")
 98    
 99    # 5. Create index files.
100    mdsys.make_sys_ndx(backbone_atoms=["BB", "BB1", "BB3"])

103def setup_cg_protein_membrane(sysdir, sysname):
104    """Sets up a coarse-grained protein/membrane system.

111        sysname (str): Name of the MD system.

113    mdsys = gmxSystem(sysdir, sysname)
114    mdsys.prepare_files()
115    mdsys.sort_input_pdb(f"{sysname}.pdb")
116    mdsys.clean_pdb_gmx(in_pdb=mdsys.inpdb, clinput="8\n 7\n", ignh="no", renum="yes")
117    mdsys.split_chains()
118    mdsys.clean_chains_gmx(clinput="8\n 7\n", ignh="yes", renum="yes")
119    mdsys.get_go_maps(append=True)
120    mdsys.martinize_proteins_go(go_eps=10.0, go_low=0.3, go_up=1.0, p="backbone", pf=500, append=True)
121    mdsys.martinize_rna(ef=200, el=0.3, eu=1.2, p="backbone", pf=500, append=True)
122    mdsys.make_martini_topology_file(add_resolved_ions=False, prefix="chain")
123    mdsys.make_cgpdb_file(bt="dodecahedron", d="1.2")
124    solvent = os.path.join(mdsys.wdir, "water.gro")

126    mdsys.add_bulk_ions(conc=0.15, pname="NA", nname="CL")

130def md(sysdir, sysname, runname, ntomp):
131    """Runs a complete MD simulation.
132
133    This function performs an MD simulation by executing the energy minimization,
134    equilibration, and production MD steps in sequence.
135
136    Args:
137        sysdir (str): Base directory for the system.
138        sysname (str): Name of the MD system.
139        runname (str): Name for the MD run.
140        ntomp (int): Number of CPU threads to use.
141    """
142    mdrun = MDRun(sysdir, sysname, runname)
143    mdrun.prepare_files()
144    # Choose appropriate mdp files.
145    em_mdp = os.path.join(mdrun.mdpdir, "em.mdp")
146    eq_mdp = os.path.join(mdrun.mdpdir, "eq.mdp")
147    md_mdp = os.path.join(mdrun.mdpdir, "md.mdp")
148    mdrun.empp(f=em_mdp)
149    mdrun.mdrun(deffnm="em", ntomp=ntomp)
150    mdrun.eqpp(f=eq_mdp, c="em.gro", r="em.gro", maxwarn=10)
151    mdrun.mdrun(deffnm="eq", ntomp=ntomp)
152    mdrun.mdpp(f=md_mdp, c="eq.gro", r="eq.gro")
153    mdrun.mdrun(deffnm="md", ntomp=ntomp)

156def extend(sysdir, sysname, runname, ntomp):
157    """Extends an ongoing MD simulation.
158
159    Args:
160        sysdir (str): Base directory for the system.
161        sysname (str): Name of the MD system.
162        runname (str): Name of the MD run to extend.
163        ntomp (int): Number of CPU threads to use.
164    """
165    mdsys = gmxSystem(sysdir, sysname)
166    mdrun = mdsys.initmd(runname)
167    mdrun.mdrun(deffnm="md", cpi="md.cpt", ntomp=ntomp, nsteps=-2)

182def trjconv(sysdir, sysname, runname, mode="solu", fit="rot+trans", **kwargs):
183    """Converts trajectory files using GROMACS trjconv.
184
185    Args:
186        sysdir (str): Base directory for the system.
187        sysname (str): Name of the MD system.
188        runname (str): Name for the MD run.
189        mode (str, optional): Mode to use for conversion ("solu" or "bb").
190        fit (str, optional): Fitting method to apply (default: "rot+trans").
191        **kwargs: Additional parameters for trjconv.
192    """
193    kwargs.setdefault("b", 0)      # in ps
194    kwargs.setdefault("dt", 1000)  # in ps
195    kwargs.setdefault("e", 1000000)  # in ps
196    mdrun = MDRun(sysdir, sysname, runname)
197    if mode == "solu":
198        k = 1
199    elif mode == "bb":
200        k = 2
201    if fit:
202        mdrun.trjconv(clinput="0\n0\n", s="mdc.pdb", f="mdc.trr", o="mdv.pdb", dt=10000)
203    clean_dir(mdrun.rundir)

206def rms_analysis(sysdir, sysname, runname, **kwargs):
207    """Performs RMSF and RMSD analysis on the MD run.
208
209    Args:
210        sysdir (str): Base directory for the system.
211        sysname (str): Name of the MD system.
212        runname (str): Name for the MD run.
213        **kwargs: Additional parameters for analysis (e.g., time range, dt).
214
215    This function calculates RMSF and RMSD using GROMACS commands.
216    """
217    kwargs.setdefault("b", 50000)   # in ps
218    kwargs.setdefault("dt", 1000)     # in ps
219    kwargs.setdefault("e", 10000000)  # in ps
220    mdrun = MDRun(sysdir, sysname, runname)
221    mdrun.rmsf(clinput="2\n 2\n", s=mdrun.str, f=mdrun.trj, n=mdrun.sysndx, fit="yes", res="yes", **kwargs)
222    mdrun.rmsd(clinput="2\n 2\n", s=mdrun.str, f=mdrun.trj, n=mdrun.sysndx, fit="rot+trans", **kwargs)

243def cov_analysis(sysdir, sysname, runname):
244    """Performs covariance and perturbation matrix analysis on the MD run.
245
246    Args:
247        sysdir (str): Base directory for the system.
248        sysname (str): Name of the MD system.
249        runname (str): Name for the MD run.
250
251    This function computes covariance matrices, perturbation matrices,
252    and calculates DFI/DCI values.
253    """
254    mdrun = MDRun(sysdir, sysname, runname)
255    u = mda.Universe(mdrun.str, mdrun.trj, in_memory=True)
256    ag = u.atoms.select_atoms("name CA or name P or name C1'")
257    if not ag:
258        ag = u.atoms.select_atoms("name BB or name BB1 or name BB3")
259    mdrun.get_covmats(u, ag, sample_rate=1, b=50000, e=1000000, n=4, outtag="covmat")
260    mdrun.get_pertmats()
261    mdrun.get_dfi(outtag="dfi")
262    mdrun.get_dci(outtag="dci", asym=False)
263    mdrun.get_dci(outtag="asym", asym=True)

266def tdlrt_analysis(sysdir, sysname, runname):

268
269    Args:
270        sysdir (str): Base directory for the system.
271        sysname (str): Name of the MD system.
272        runname (str): Name for the MD run.
273
274    This function calculates the position-velocity correlation function using a GPU method.
275    """
276    mdrun = MDRun(sysdir, sysname, runname)
277    b = 0
278    e = 100000
279    sample_rate = 1
280    ntmax = 1000
281    tag = "pv"
282    corr_file = os.path.join(mdrun.lrtdir, f"corr_{tag}.npy")
283    u = mda.Universe(mdrun.str, mdrun.trj, in_memory=True)
284    ag = u.atoms
285    positions = io.read_positions(u, ag, sample_rate=sample_rate, b=b, e=e)
286    velocities = io.read_velocities(u, ag, sample_rate=sample_rate, b=b, e=e)
287    corr = lrt.ccf(positions, velocities, ntmax=ntmax, n=5, mode="gpu", center=True)

310def get_td_averages(sysdir, sysname, loop=True, fname="corr_pv.npy"):
311    """Calculates time-dependent averages from analysis files.
312
313    Args:
314        sysdir (str): Base directory for the system.
315        sysname (str): Name of the MD system.
316        loop (bool, optional): If True, processes files sequentially. Default is True.
317        fname (str, optional): Filename pattern to average (default: "corr_pv.npy").
318
319    Returns:
320        numpy.ndarray: The time-dependent average.
321    """
322    mdsys = gmxSystem(sysdir, sysname)
323    print("Getting averages", file=sys.stderr)
324    files = io.pull_files(mdsys.mddir, fname)
325    if loop:
326        print(f"Processing {files[0]}", file=sys.stderr)
327        average = np.load(files[0])
328        for f in files[1:]:
329            print(f"Processing {f}", file=sys.stderr)
330            arr = np.load(f)
331            average += arr
332        average /= len(files)
333    else:
334        arrays = [np.load(f) for f in files]
335        average = np.average(arrays, axis=0)
336    np.save(os.path.join(mdsys.datdir, fname), average)
337    print("Done!", file=sys.stderr)
338    return average
